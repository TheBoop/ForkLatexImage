\documentclass{article}
\usepackage[utf8]{inputenc}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric, arrows}

\begin{document}

\tikzstyle{pid} = [rectangle,rounded corners, minimum width=3cm, minimum height=1cm, text width=3cm,text centered, draw=black]
\tikzstyle{action} = [rectangle,minimum width=3cm, minimum height=1cm, text centered, draw=black]
\tikzstyle{comment} = [rectangle,rounded corners,text centered,draw=black, minimum height=0.8cm,fill=yellow!10]
\tikzstyle{state} = [circle, minimum size=1cm, thick ,draw=black,text centered]
\tikzstyle{arrow} = [thick,->,>=stealth,scale=2]

\hspace*{-3cm}\begin{tikzpicture}[
node distance=2cm,
level distance=2cm,
level 1/.style = {sibling distance=8cm},
level 2/.style = {sibling distance=4cm},
level 3/.style = {sibling distance=4cm},
levelCom/.style = {level distance=3cm}
]

\node(start)[pid, fill=red!10]{Parent};
\node(fork1)[action, below of=start]{int $PID1 = fork()$}
child{
    node(child1)[pid, fill=blue!10]{Child\\ $PID1==0$}
    %level2 Children
    child{ node(fork21)[action]{int $PID2 = fork()$ }
    %level3 Children
    child{node(child21)[pid, fill=blue!10]{Children\\ $PID1==0$\\ $PID2==0$}
        child[levelCom]{node(exit1)[state]{exit}
            child[level distance=1.5cm]{node(empty1){} edge from parent[arrow]}
            }
        }
            child{node(parent21)[pid, fill=green!10]{Parent\\ $PID1==0$\\ $PID2>0$}}
            }
        }
        child{
            node(parent1)[pid, fill=green!10]{Parent\\ $PID1>0$}
            %level2 Children
            child{ node(fork22)[action]{int $PID2 = fork()$ }
            %level3 Children
            child{node(child22)[pid, fill=blue!10]{Children\\ $PID1>0$\\ $PID2==0$}
                child[levelCom]{node(exit2)[state]{exit}
                child[level distance=1.5cm]{node(empty2){} edge from parent[arrow]}
            }
        }
        child{node(parent22)[pid, fill=green!10]{Parent\\ $PID1>0$\\ $PID2>0$}
            child[levelCom]{node(wait)[state]{wait()}
                child[level distance=1.5cm]{node(emptyWait){}}
                }
        }
    }
}
;
\node(shell)[right of=wait]{};

\draw [arrow] (start) -- (fork1);
\draw [arrow] (fork1) -- (child1);
\draw [arrow] (fork1) -- (parent1);
\draw [arrow] (child1) -- (fork21);
\draw [arrow] (parent1) -- (fork22);
\draw [arrow] (fork21) -- (child21);
\draw [arrow] (fork21) -- (parent21);
\draw [arrow] (fork22) -- (child22);
\draw [arrow] (fork22) -- (parent22);
\draw [arrow] (child21) -- node[anchor=east,comment]{execvp(arg[0],argv)} (exit1);
\draw [arrow] (child22) -- node[anchor=east,comment]{execvp(arg[0],argv)} (exit2);
\draw [arrow] (parent22) -- (wait);
\draw [arrow] (empty1) -- node[anchor=south,comment]{zombie processes} (empty2);
\draw [arrow] (empty2) -- node[anchor=south,comment,text width=2cm]{zombie processes} (emptyWait);
\draw [arrow] (emptyWait) -- (wait);
\draw [arrow] (wait) -- node[anchor=south,comment,fill=red!10]{\textbf{shell}} (shell);


\end{tikzpicture}

\end{document}
